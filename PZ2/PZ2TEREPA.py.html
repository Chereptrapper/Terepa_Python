n = input('Введите количество секунд (максимальное 86400): ') #Получение количества секунд
while type(n) != int: #Обработка исключений
    try:
        n = int(n)
    except ValueError:
        print('Вы ввели не число')
        n = input('Введите количество секунд (максимальное 86400): ')
x = n % 3600 # Получение секунд последнего часа
print(f'Секунды с последнего часа: {x}') # Вывод результата
